	$(a=b) \vdash (b=a)$
	\begin{enumerate}
		\item $(a=b) \supset ((a=c) \supset (b=c))$ — постулаты арифметики, \circled{16}
		\item $(a=b) \supset \forall{c} ((a=c) \supset (b=c))$ — постулаты исчисления предикатов, \circled{9}
		\item $\forall{c}((a=c)\supset (b=c))$ — исчиление высказываний, \circled{2}, из исходной формулы и 2 
		\item $\forall{c}((a=c)\supset (b=c))\supset((a=a) \supset (b=a)) $ — исчисление предикатов, \circled{10}, $t=a$
		\item $(a=a) \supset (b=a)$ — исчиление высказываний, \circled{2}, из 3 и 4 
		\item $b=a $ — исчиление высказываний, \circled{2}, из $a=a$ (непосредственное следствие из аксиом) и 5
	\end{enumerate}

	\bigskip
	
	$(a=b), (b=c) \vdash (a=c)$
	\begin{enumerate}
		\item $(b=a) \supset ((b=d) \supset (a=d))$ — \circled{16}
		\item $(b=a) \supset \forall{d} ((b=d) \supset (a=d))$ — \circled{9}
		\item $\forall{d} ((b=d) \supset (a=d))$ — \circled{2}, из $(a=b) \vdash (b=a)$ и 2 
		\item $\forall{d} ((b=d) \supset (a=d)) \supset ((b=c) \supset (a=c))$ — \circled{10}, $ t=c$
		\item $(b=c) \supset (a=c)$ — \circled{2}, из 3 и 4
		\item $a=c$ — \circled{2}, из исходной формулы и 5
	\end{enumerate}

	\bigskip
	
	По теореме о дедукции:
	\begin{enumerate}
		\item $(a=b), (b=c) \vdash (a=c)$
		\item $(a=b) \vdash (b=c) \supset (a=c)$
		\item $\vdash (a=b) \supset ((b=c) \supset (a=c))$
	\end{enumerate}
